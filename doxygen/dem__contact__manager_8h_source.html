27/**
28 * @brief Manage the numerous of contact detection search operations in the DEM.
29 *
30 * This class mostly calls proper functions in regards the type of contacts for
31 * the contact detection and updates of data containers.
32 */
33template <int dim>
34class DEMContactManager
35{
36public:
37 /**
38 * @brief Execute functions to find the lists of cell neighbors of the active
39 * cells in the triangulation.
40 *
41 * For contacts of particles, the neighbor search excludes the repetitions of
42 * neighbors: B is neighbor of A but A will not be neighbor of B. For contacts
43 * of particles with floating meshes, the neighbor search includes all
44 * neighbors. These floating mesh contacts searches need all the particles
45 * located in all the neighbor cells of the main cell to detect
46 * collisions with the floating mesh.
47 *
48 * @param[in] triangulation The triangulation to access the information of the
49 * cells.
50 * @param[in] periodic_boundaries_cells_information Information of periodic
51 * cells used if periodic boundaries are enabled (next parameter).
52 */
53 void
54 execute_cell_neighbors_search(
55 const parallel::distributed::Triangulation<dim> &triangulation,
56 const typename DEM::dem_data_structures<dim>::periodic_boundaries_cells_info
57 periodic_boundaries_cells_information);
58
59 /**
60 * @brief Execute functions to clear and update the neighbor lists of all the
61 * active cells in the triangulation.
62 *
63 * @param[in] triangulation The triangulation to access the information of the
64 * cells.
65 * @param[in] periodic_boundaries_cells_information Information of periodic
66 * cells used if periodic boundaries are enabled (next parameter).
67 */
68 void
69 update_cell_neighbors(
70 const parallel::distributed::Triangulation<dim> &triangulation,
71 const typename DEM::dem_data_structures<dim>::periodic_boundaries_cells_info
72 periodic_boundaries_cells_information)
73 {
74 cells_local_neighbor_list.clear();
75 cells_ghost_neighbor_list.clear();
76 cells_local_periodic_neighbor_list.clear();
77 cells_ghost_periodic_neighbor_list.clear();
78 cells_ghost_local_periodic_neighbor_list.clear();
79 total_neighbor_list.clear();
80
81 execute_cell_neighbors_search(triangulation,
82 periodic_boundaries_cells_information);
83 }
84
85 /**
86 * @brief Execute functions to update the contact data containers with the new
87 * contact pairs.
88 *
89 * Call proper functions to remove contact repetitions and to add new contact
90 * pairs to the contact containers when particles are exchanged between
91 * processors after the fine search.
92 * Contact pairs are local particle-particle, ghost particle-particle,
93 * particle-wall, particle-floating wall contacts and particle-floating mesh
94 * contacts.
95 */
96 void
97 update_contacts();
98
99 /**
100 * @breif Execute functions to update the particle iterators in local-local
101 * contact containers.
102 *
103 * This is essential since sort_particles_into_subdomains_and_cells() and
104 * exchange_ghost_particles() functions change the particle iterators
105 * everytime they are called.
106 *
107 * @param[in] particle_handler Storage of particles and their accessor
108 * functions.
109 */
110 void
111 update_local_particles_in_cells(
112 const Particles::ParticleHandler<dim> &particle_handler);
113
114 /**
115 * @brief Execute the particle-particle broad searches with adaptive sparse
116 * contacts.
117 *
118 * Calls proper functions to find the candidates of local and ghost
119 * particle-particle contact pairs and the periodic particle-particle contacts
120 * if required. These contact pairs will be used in the fine search step to
121 * investigate if they are in contact.
122 * It checks if the adaptive sparse contacts is enabled and use proper
123 * functions.
124 *
125 * @param[in,out] particle_handler Storage of particles and their accessor
126 * functions.
127 * @param[in] sparse_particle_contact_object Allow to check the mobility
128 * status of cells.
129 */
130 void
131 execute_particle_particle_broad_search(
132 dealii::Particles::ParticleHandler<dim> &particle_handler,
133 const AdaptiveSparseContacts<dim> &sparse_particle_contact_object);
134
135 /**
136 * @brief Execute the particle-wall broad searches with adaptive sparse
137 * contacts.
138 *
139 * Calls proper functions to find the candidates of particle-wall contact
140 * pairs and the particle-floating wall or particle-floating mesh if required.
141 * These contact pairs will be used in the fine search step to investigate if
142 * they are in contact.
143 * It checks if the adaptive sparse contacts is enabled and use proper
144 * functions.
145 *
146 * @param[in] particle_handler Storage of particles and their accessor
147 * functions.
148 * @param[in] boundary_cells_object Information of the boundary cells and
149 * faces.
150 * @param[in] solid_surfaces_mesh_info Mapping of solid surfaces meshes.
151 * @param[in] floating_wall Properties of the floating walls.
152 * @param[in] simulation_time Current simulation time.
153 * @param[in] sparse_particle_contact_object Allow to check the mobility
154 * status of cells.
155 */
156 void
157 execute_particle_wall_broad_search(
158 const Particles::ParticleHandler<dim> &particle_handler,
159 BoundaryCellsInformation<dim> &boundary_cell_object,
160 const typename dem_data_structures<dim>::solid_surfaces_mesh_information
161 solid_surfaces_mesh_info,
162 const Parameters::Lagrangian::FloatingWalls<dim> &floating_walls,
163 const double simulation_time,
164 const AdaptiveSparseContacts<dim> &sparse_particle_contact_object);
165
166 /**
167 * @brief Execute the particle-particles fine searches.
168 *
169 * Executes functions that update the particle contacts pairs containers and
170 * compute the contact information of the collision pairs.
171 *
172 * @param[in] neighborhood_threshold Threshold value of contact detection.
173 */
174 void
175 execute_particle_particle_fine_search(const double neighborhood_threshold);
176
177 /**
178 * @brief Execute the particle-wall fine searches.
179 *
180 * Executes functions that update the particle-wall contacts pairs containers
181 * and compute the contact information of the collision particle-wall.
182 *
183 * @param[in] floating_wall Properties of the floating walls.
184 * @param[in] simulation_time Current simulation time.
185 * @param[in] neighborhood_threshold Threshold value of contact detection.
186 */
187 void
188 execute_particle_wall_fine_search(
189 const Parameters::Lagrangian::FloatingWalls<dim> &floating_walls,
190 const double simulation_time,
191 const double neighborhood_threshold);
192
193 /**
194 * @brief Set the constant periodic offset for the periodic boundaries. If
195 * they are no periodic boundaries, the default offset is zeros;

198 * periodic boundaries.
199 */
200 inline void
201 set_periodic_offset(const Tensor<1, dim> &offset)
202 {
203 this->periodic_offset = offset;
