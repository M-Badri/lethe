6 template <int dim>
7 void
8 VOFCurvatureProjection<dim>::assemble_system_matrix_and_rhs()
9 {
10  // Reinitialize system matrix and right-hand side (rhs)
11  this->system_matrix = 0;
12  this->system_rhs = 0;
13 
14  // Get phase gradient L2 projection DoFHandler
15  const DoFHandler<dim> *dof_handler_phase_fraction_gradient =
16  this->subequations_interface->get_dof_handler(
17  VOFSubequationsID::phase_gradient_projection);
18 
19  // Initialize FEValues for curvature and phase fraction gradient projection
20  FEValues<dim> fe_values_curvature_projection(*this->mapping,
21  *this->fe,
22  *this->cell_quadrature,
23  update_values |
24  update_gradients |
25  update_JxW_values);
26  FEValues<dim> fe_values_phase_gradient_projection(
27  *this->mapping,
28  dof_handler_phase_fraction_gradient->get_fe(),
29  *this->cell_quadrature,
30  update_values);
31 
32  // Initialize size of arrays
33  const double n_q_points =
34  fe_values_curvature_projection.get_quadrature().size();
35  const double n_dofs_per_cell =
36  fe_values_curvature_projection.get_fe().n_dofs_per_cell();
37 
38  // Initialize local matrix and rhs
39  FullMatrix<double> local_matrix(n_dofs_per_cell, n_dofs_per_cell);
40  Vector<double> local_rhs(n_dofs_per_cell);
41 
42  std::vector<types::global_dof_index> local_dof_indices(n_dofs_per_cell);
43 
44  // Extractor for phase fraction gradient vector
45  FEValuesExtractors::Vector phase_fraction_gradients(0);
46 
47  // Initialize solution arrays
48  std::vector<double> present_curvature_projection_values(n_q_points);
49  std::vector<Tensor<1, dim>> present_curvature_projection_gradients(
50  n_q_points);
51  std::vector<Tensor<1, dim>> present_phase_gradient_projection_values(
52  n_q_points);
53 
54  // Initialize shape function arrays
55  std::vector<double> phi(n_dofs_per_cell);
56  std::vector<Tensor<1, dim>> grad_phi(n_dofs_per_cell);
57 
58  // Get the diffusion factor
59  const double diffusion_factor =
60  this->simulation_parameters.multiphysics.vof_parameters
61  .surface_tension_force.curvature_diffusion_factor;
62 
63  // Loop over phase gradient projection cells
64  for (const auto &cell : this->dof_handler.active_cell_iterators())
65  {
66  if (cell->is_locally_owned())
67  {
68  // Reinitialize local linear system
69  local_matrix = 0;
70  local_rhs = 0;
71 
72  // Get phase gradient projection cell iterator
73  typename DoFHandler<dim>::active_cell_iterator
74  phase_gradient_projection_cell(&(*this->triangulation),
75  cell->level(),
76  cell->index(),
77  dof_handler_phase_fraction_gradient);
78 
79  // Reinitialize FEValues with corresponding cells
80  fe_values_curvature_projection.reinit(cell);
81  fe_values_phase_gradient_projection.reinit(
82  phase_gradient_projection_cell);
83 
84  // Get vector of Jacobi determinant times the quadrature weights
85  std::vector<double> JxW_vec =
86  fe_values_curvature_projection.get_JxW_values();
87 
88  // Compute cell size
89  auto &fe_curvature_projection =
90  fe_values_curvature_projection.get_fe();
91  const double h =
92  compute_cell_diameter<dim>(compute_cell_measure_with_JxW(JxW_vec),
93  fe_curvature_projection.degree);
94 
95  // Get projected curvature values, gradients and projected phase
96  // fraction gradient values
97  fe_values_curvature_projection.get_function_values(
98  this->present_solution, present_curvature_projection_values);
99  fe_values_curvature_projection.get_function_gradients(
100  this->present_solution, present_curvature_projection_gradients);
101  fe_values_phase_gradient_projection[phase_fraction_gradients]
102  .get_function_values(
103  *this->subequations_interface->get_solution(
104  VOFSubequationsID::phase_gradient_projection),
105  present_phase_gradient_projection_values);
106 
107  // Set tolerance to avoid division by zero
108  const double tolerance = 1e-12;
109 
110  // Loop over quadrature points
111  for (unsigned int q = 0; q < n_q_points; ++q)
112  {
113  // Shape functions
114  for (unsigned int k = 0; k < n_dofs_per_cell; ++k)
115  {
116  phi[k] = fe_values_curvature_projection.shape_value(k, q);
117  grad_phi[k] = fe_values_curvature_projection.shape_grad(k, q);
118  }
119 
120  // Compute projected phase gradient norm
121  const Tensor<1, dim> projected_vof_phase_gradient =
122  present_phase_gradient_projection_values[q];
123  const double projected_vof_phase_gradient_norm =
124  projected_vof_phase_gradient.norm() + tolerance;
125 
126  // Local linear system assembly
127  for (unsigned int i = 0; i < n_dofs_per_cell; ++i)
128  {
129  // Assemble local matrix
130  for (unsigned int j = 0; j < n_dofs_per_cell; ++j)
131  {
132  local_matrix(i, j) +=
133  (phi[i] * phi[j] +
134  h * h * diffusion_factor *
135  scalar_product(grad_phi[i], grad_phi[j])) *
136  JxW_vec[q];
137  }
138  // Assemble local right-hand side (rhs)
139  local_rhs(i) += grad_phi[i] *
140  (projected_vof_phase_gradient /
141  projected_vof_phase_gradient_norm) *
142  JxW_vec[q];
143  }
144  }
145 
146  // Distribute the local contributions to the global system
147  cell->get_dof_indices(local_dof_indices);
148  this->constraints.distribute_local_to_global(local_matrix,
149  local_rhs,
150  local_dof_indices,
151  this->system_matrix,
152  this->system_rhs);
153  }
154  }
155  this->system_matrix.compress(VectorOperation::add);
156  this->system_rhs.compress(VectorOperation::add);
157 }
