13template <int dim>
14void
15Visualization<dim>::build_patches(
16 dealii::Particles::ParticleHandler<dim> &particle_handler,
17 std::vector<std::pair<std::string, int>> properties)
18{
19 // Adding ID to properties vector for visualization
20 properties.insert(properties.begin(), std::make_pair("ID", 1));
21
22 // Defining properties for writing
23 this->properties_to_write.assign(properties.begin(),
24 properties.begin() +
25 DEM::get_number_properties());
26
27 // Defining property field position
28 int field_position = 0;
29 // Iterating over properties
30 for (auto properties_iterator = properties_to_write.begin();
31 properties_iterator != properties_to_write.end();
32 ++properties_iterator, ++field_position)
33 {
34 // Get the property field name
35 const std::string field_name = properties_iterator->first;
36
37 // Number of components of the corresponding property
38 const unsigned components_number = properties_iterator->second;
39
40 // Check to see if the property is a vector
41 if (components_number == dim)
42 {
43 vector_datasets.emplace_back(std::make_tuple(
44 field_position,
45 field_position + components_number - 1,
46 field_name,
47 DataComponentInterpretation::component_is_part_of_vector));
48 }
49 dataset_names.emplace_back(field_name);
50 }
51
52 // Building the patch data
53 patches.resize(particle_handler.n_locally_owned_particles());
54
55 typename dealii::Particles::ParticleHandler<dim>::particle_iterator particle =
56 particle_handler.begin();
57
58 // Looping over particle to get the properties from the particle_handler
59 for (unsigned int i = 0; particle != particle_handler.end(); ++particle, ++i)
60 {
61 // Particle location
62 patches[i].vertices[0] = particle->get_location();
63 patches[i].patch_index = i;
64 patches[i].data.reinit(DEM::get_number_properties(), 1);
65
66 // ID and other properties
67 if (particle->has_properties())
68 {
69 // Calculating force for visualization
70 auto particle_properties = particle->get_properties();
71
72 // Adding ID to patches
73 patches[i].data(0, 0) = particle->get_id();
74
75 for (unsigned int property_index = 1;
76 property_index < DEM::get_number_properties();
77 ++property_index)
78 patches[i].data(property_index, 0) =
79 particle_properties[property_index - 1];
80 }
81 }
82}

84template <int dim>
85void
86Visualization<dim>::print_xyz(
87 dealii::Particles::ParticleHandler<dim> &particle_handler,
88 const MPI_Comm &mpi_communicator,
89 const ConditionalOStream &pcout)
90{
91 pcout << "id, type, dp, x, y, z " << std::endl;
92 // Agressively force synchronization of the header line
93 usleep(500);
94 MPI_Barrier(mpi_communicator);
95 usleep(500);
96 MPI_Barrier(mpi_communicator);
97
98 std::map<int, Particles::ParticleIterator<dim>> global_particles;
99 unsigned int current_id, current_id_max = 0;
100
101 // Mapping of all particles & find the max id on current processor
102 for (auto particle = particle_handler.begin();
103 particle != particle_handler.end();
104 ++particle)
105 {
106 current_id = particle->get_id();
107 current_id_max = std::max(current_id, current_id_max);
108
109 global_particles.insert({current_id, particle});
110 }
111
112 // Find global max particle index
113 unsigned int id_max = Utilities::MPI::max(current_id_max, mpi_communicator);
114
115 // Print particle info one by one in ascending order
116 for (unsigned int i = 0; i <= id_max; i++)
117 {
118 for (auto &iterator : global_particles)
119 {
120 unsigned int id = iterator.first;
121 if (id == i)
122 {
123 auto particle = iterator.second;
124 auto particle_properties = particle->get_properties();
125 auto particle_location = particle->get_location();
126
127 std::cout << std::fixed << std::setprecision(0) << id << " "
128 << std::setprecision(0)
129 << particle_properties[DEM::PropertiesIndex::type]
130 << " " << std::setprecision(5)
131 << particle_properties[DEM::PropertiesIndex::dp] << " "
132 << std::setprecision(4) << particle_location
133 << std::endl;
134 }
135 }
136 usleep(500);
137 MPI_Barrier(mpi_communicator);
138 }
139}

141template <int dim>
142void
143Visualization<dim>::print_intermediate_format(
144 const Vector<float> &data_to_print,
145 const DoFHandler<dim> &background_dh,
146 const MPI_Comm &mpi_communicator)
147{
148 unsigned int n_mpi_processes(
149 Utilities::MPI::n_mpi_processes(mpi_communicator));
150 unsigned int this_mpi_process(
151 Utilities::MPI::this_mpi_process(mpi_communicator));
152
153 // Attach the data to the background mesh
154 DataOut<dim> data_out;
155 data_out.attach_dof_handler(background_dh);
156 data_out.add_data_vector(data_to_print,
157 "print_from_processor_" +
158 Utilities::int_to_string(this_mpi_process),
159 DataOut<dim>::type_cell_data);
160 data_out.build_patches();
161
162 std::stringstream out;
163
164 // Add data in deal.II intermediate format to the stringstream buffer for each
165 // processor in order
166 for (unsigned int processor_number = 0; processor_number < n_mpi_processes;
167 ++processor_number)
168 {
169 usleep(100);
170 MPI_Barrier(mpi_communicator);
171 if (processor_number == this_mpi_process)
172 {
173 // Generate a string stream buffer to store the output
174 data_out.write_deal_II_intermediate(out);
175
176 // Print in terminal but remove part of the header since it
177 // contains some deal.II version information that may change
178 std::string line;
179 unsigned int counter = 0;
180 while (std::getline(out, line))
181 {
182 if (counter++ > 4)
183 std::cout << line << std::endl;
184 }
185 }
186 }
187}
