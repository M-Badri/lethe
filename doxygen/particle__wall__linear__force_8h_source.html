107 /**
108 * @brief Carries out calculation of the rolling resistance torque using the constant model
109 *
110 * @param particle_one_properties Particle one properties
111 * @param particle_two_properties Particle two properties
112 * @param effective_rolling_friction_coefficient Effective rolling friction coefficient
113 * @param normal_force_norm Normal force norm
114 *
115 * @return rolling resistance torque
116 */
117 inline Tensor<1, 3>
118 constant_resistance(const ArrayView<const double> &particle_properties,
119 const double effective_rolling_friction_coefficient,
120 const double normal_force_norm,
121 const Tensor<1, 3> & /*normal_contact_vector*/)
122 {
123 // Getting the angular velocity of particle in the vector format
124 Tensor<1, 3> angular_velocity;
125 for (int d = 0; d < 3; ++d)
126 {
127 angular_velocity[d] = particle_properties[PropertiesIndex::omega_x + d];
128 }
129
130 // Calculation of particle-wall angular velocity (norm of the
131 // particle angular velocity)
132 Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
133
134 double omega_value = angular_velocity.norm();
135 if (omega_value != 0)
136 {
137 particle_wall_angular_velocity = angular_velocity / omega_value;
138 }
139
140 // Calcualation of rolling resistance torque
141 Tensor<1, 3> rolling_resistance_torque =
142 -effective_rolling_friction_coefficient *
143 (particle_properties[PropertiesIndex::dp] * 0.5) * normal_force_norm *
144 particle_wall_angular_velocity;
145
146 return rolling_resistance_torque;
147 }

149 /**
150 * @brief Carries out calculation of the rolling resistance torque using the viscous model
151 *
152 * @param particle_one_properties Particle one properties
153 * @param particle_two_properties Particle two properties
154 * @param effective_rolling_friction_coefficient Effective rolling friction coefficient
155 * @param normal_force_norm Normal force norm
156 *
157 * @return rolling resistance torque
158 */
159 inline Tensor<1, 3>
160 viscous_resistance(const ArrayView<const double> &particle_properties,
161 const double effective_rolling_friction_coefficient,
162 const double normal_force_norm,
163 const Tensor<1, 3> &normal_contact_vector)
164 {
165 // Getting the angular velocity of particle in the vector format
166 Tensor<1, 3> angular_velocity;
167 for (int d = 0; d < 3; ++d)
168 {
169 angular_velocity[d] = particle_properties[PropertiesIndex::omega_x + d];
170 }
171
172 // Calculation of particle-wall angular velocity (norm of the
173 // particle angular velocity)
174 Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
175
176 double omega_value = angular_velocity.norm();
177 if (omega_value != 0)
178 {
179 particle_wall_angular_velocity = angular_velocity / omega_value;
180 }
181
182 Tensor<1, 3> v_omega =
183 cross_product_3d(angular_velocity,
184 particle_properties[PropertiesIndex::dp] * 0.5 *
185 normal_contact_vector);
186
187 // Calculation of rolling resistance torque
188 Tensor<1, 3> rolling_resistance_torque =
189 -effective_rolling_friction_coefficient *
190 particle_properties[PropertiesIndex::dp] * 0.5 * normal_force_norm *
191 v_omega.norm() * particle_wall_angular_velocity;
192
193 return rolling_resistance_torque;
194 }
