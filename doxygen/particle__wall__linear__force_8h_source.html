103 /**
104 * @brief Carries out calculation of the rolling resistance torque using the constant model
105 *
106 * @param particle_one_properties Particle one properties
107 * @param particle_two_properties Particle two properties
108 * @param effective_rolling_friction_coefficient Effective rolling friction coefficient
109 * @param normal_force_norm Normal force norm
110 *
111 * @return rolling resistance torque
112 */
113 inline Tensor<1, 3>
114 constant_resistance(const ArrayView<const double> &particle_properties,
115 const double effective_rolling_friction_coefficient,
116 const double normal_force_norm,
117 const Tensor<1, 3> & /*normal_contact_vector*/)
118 {
119 // Getting the angular velocity of particle in the vector format
120 Tensor<1, 3> angular_velocity;
121 for (int d = 0; d < 3; ++d)
122 {
123 angular_velocity[d] =
124 particle_properties[DEM::PropertiesIndex::omega_x + d];
125 }
126
127 // Calculation of particle-wall angular velocity (norm of the
128 // particle angular velocity)
129 Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
130
131 double omega_value = angular_velocity.norm();
132 if (omega_value != 0)
133 {
134 particle_wall_angular_velocity = angular_velocity / omega_value;
135 }
136
137 // Calcualation of rolling resistance torque
138 Tensor<1, 3> rolling_resistance_torque =
139 -effective_rolling_friction_coefficient *
140 (particle_properties[DEM::PropertiesIndex::dp] * 0.5) *
141 normal_force_norm * particle_wall_angular_velocity;
142
143 return rolling_resistance_torque;
144 }

146 /**
147 * @brief Carries out calculation of the rolling resistance torque using the viscous model
148 *
149 * @param particle_one_properties Particle one properties
150 * @param particle_two_properties Particle two properties
151 * @param effective_rolling_friction_coefficient Effective rolling friction coefficient
152 * @param normal_force_norm Normal force norm
153 *
154 * @return rolling resistance torque
155 */
156 inline Tensor<1, 3>
157 viscous_resistance(const ArrayView<const double> &particle_properties,
158 const double effective_rolling_friction_coefficient,
159 const double normal_force_norm,
160 const Tensor<1, 3> &normal_contact_vector)
161 {
162 // Getting the angular velocity of particle in the vector format
163 Tensor<1, 3> angular_velocity;
164 for (int d = 0; d < 3; ++d)
165 {
166 angular_velocity[d] =
167 particle_properties[DEM::PropertiesIndex::omega_x + d];
168 }
169
170 // Calculation of particle-wall angular velocity (norm of the
171 // particle angular velocity)
172 Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
173
174 double omega_value = angular_velocity.norm();
175 if (omega_value != 0)
176 {
177 particle_wall_angular_velocity = angular_velocity / omega_value;
178 }
179
180 Tensor<1, 3> v_omega =
181 cross_product_3d(angular_velocity,
182 particle_properties[DEM::PropertiesIndex::dp] * 0.5 *
183 normal_contact_vector);
184
185 // Calculation of rolling resistance torque
186 Tensor<1, 3> rolling_resistance_torque =
187 -effective_rolling_friction_coefficient *
188 particle_properties[DEM::PropertiesIndex::dp] * 0.5 * normal_force_norm *
189 v_omega.norm() * particle_wall_angular_velocity;
190
191 return rolling_resistance_torque;
192 }
