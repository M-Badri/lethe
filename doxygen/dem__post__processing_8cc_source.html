10 template <int dim, dem_statistic_variable var>
11 statistics
12 calculate_granular_statistics(
13 const Particles::ParticleHandler<dim> &particle_handler,
14 const MPI_Comm &mpi_communicator)
15 {
16 double total_variable = 0;
17 double max_variable = DBL_MIN;
18 double min_variable = DBL_MAX;
19
20
21 for (auto &particle : particle_handler)
22 {
23 // Get the properties of the particle
24 auto particle_properties = particle.get_properties();
25
26 double variable = 0;
27
28 if constexpr (var ==
29 dem_statistic_variable::translational_kinetic_energy)
30 {
31 // Put particle velocity in a tensor
32 Tensor<1, dim> velocity;
33 for (unsigned d = 0; d < dim; ++d)
34 velocity[d] = particle_properties[DEM::PropertiesIndex::v_x + d];
35
36 // Kinetic energy is 0.5*m*u^2
37 variable = 0.5 * particle_properties[DEM::PropertiesIndex::mass] *
38 velocity.norm_square();
39 }
40 if constexpr (var == dem_statistic_variable::rotational_kinetic_energy)
41 {
42 // Put angular velocity in a tensor
43 Tensor<1, 3> omega;
44 if constexpr (dim == 2)
45 {
46 omega[2] = particle_properties[DEM::PropertiesIndex::omega_z];
47 }
48 if constexpr (dim == 3)
49 {
50 for (unsigned d = 0; d < dim; ++d)
51 omega[d] =
52 particle_properties[DEM::PropertiesIndex::omega_x + d];
53 }
54
55 // Kinetic energy is 0.1*m*d^2 * w^2
56 variable = 0.1 * particle_properties[DEM::PropertiesIndex::mass] *
57 Utilities::fixed_power<2>(
58 particle_properties[DEM::PropertiesIndex::dp]) *
59 omega.norm_square();
60 }
61
62 if constexpr (var == dem_statistic_variable::velocity)
63 {
64 // Put particle velocity in a tensor
65 Tensor<1, dim> velocity;
66 for (unsigned d = 0; d < dim; ++d)
67 velocity[d] = particle_properties[DEM::PropertiesIndex::v_x + d];
68
69 variable = velocity.norm();
70 }
71
72 if constexpr (var == dem_statistic_variable::omega)
73 {
74 // Put angular velocity in a tensor
75 Tensor<1, 3> omega;
76 if constexpr (dim == 2)
77 {
78 omega[2] = particle_properties[DEM::PropertiesIndex::omega_z];
79 }
80 if constexpr (dim == 3)
81 {
82 for (unsigned d = 0; d < dim; ++d)
83 omega[d] =
84 particle_properties[DEM::PropertiesIndex::omega_x + d];
85 }
86
87 variable = omega.norm();
88 }
89
90 total_variable += variable;
91
92 max_variable = std::max(variable, max_variable);
93 min_variable = std::min(variable, min_variable);
94 }
95
96 total_variable = Utilities::MPI::sum(total_variable, mpi_communicator);
97 max_variable = Utilities::MPI::max(max_variable, mpi_communicator);
98 min_variable = Utilities::MPI::min(min_variable, mpi_communicator);
99
100 statistics stats;
101 stats.total = total_variable;
102 stats.max = max_variable;
103 stats.min = min_variable;
104 stats.average = total_variable / particle_handler.n_global_particles();
105
106
107 return stats;
108 }
