18template <int dim>
19void
20calculate_average_particles_velocity(
21 const parallel::distributed::Triangulation<dim> &triangulation,
22 const Particles::ParticleHandler<dim> &particle_handler,
23 Vector<double> &velocity_average_x,
24 Vector<double> &velocity_average_y,
25 Vector<double> &velocity_average_z,
26 Vector<double> &velocity_average_magnitude)
27{
28 // Iterating through the active cells in the triangulation
29 for (const auto &cell : triangulation.active_cell_iterators())
30 {
31 if (cell->is_locally_owned())
32 {
33 Tensor<1, dim> cell_velocity_average =
34 calculate_cell_average_particles_velocity(cell, particle_handler);
35
36 velocity_average_x[cell->active_cell_index()] =
37 cell_velocity_average[0];
38 velocity_average_y[cell->active_cell_index()] =
39 cell_velocity_average[1];
40
41 if constexpr (dim == 3)
42 {
43 velocity_average_z[cell->active_cell_index()] =
44 cell_velocity_average[2];
45 velocity_average_magnitude[cell->active_cell_index()] =
46 sqrt(pow(cell_velocity_average[0], 2) +
47 pow(cell_velocity_average[1], 2) +
48 pow(cell_velocity_average[2], 2));
49 }
50
51 if constexpr (dim == 2)
52 velocity_average_magnitude[cell->active_cell_index()] =
53 sqrt(pow(cell_velocity_average[0], 2) +
54 pow(cell_velocity_average[1], 2));
55 }
56 }
57}

59template <int dim>
60void
61calculate_average_granular_temperature(
62 const parallel::distributed::Triangulation<dim> &triangulation,
63 const Particles::ParticleHandler<dim> &particle_handler,
64 Vector<double> &granular_temperature_average)
65{
66 // Iterating through the active cells in the triangulation
67 for (const auto &cell : triangulation.active_cell_iterators())
68 {
69 if (cell->is_locally_owned())
70 {
71 double granular_temperature_cell(0);
72 unsigned int particles_cell_number(0);
73
74 // Looping through all the particles in the cell
75 // Particles in the cell
76 typename Particles::ParticleHandler<dim>::particle_iterator_range
77 particles_in_cell = particle_handler.particles_in_cell(cell);
78
79 const bool particles_exist_in_main_cell = !particles_in_cell.empty();
80 // Check to see if the cell has any particles
81 if (particles_exist_in_main_cell)
82 {
83 Tensor<1, dim> velocity_in_cell_average =
84 calculate_cell_average_particles_velocity(cell,
85 particle_handler);
86
87 // Initializing velocity fluctuations
88 Tensor<1, dim> cell_velocity_fluctuation_squared_sum =
89 Tensor<1, dim>();
90 Tensor<1, dim> cell_velocity_fluctuation_squared_average =
91 Tensor<1, dim>();
92
93 for (typename Particles::ParticleHandler<
94 dim>::particle_iterator_range::iterator
95 particles_in_cell_iterator = particles_in_cell.begin();
96 particles_in_cell_iterator != particles_in_cell.end();
97 ++particles_in_cell_iterator)
98 {
99 auto particle_properties =
100 particles_in_cell_iterator->get_properties();
101
102 for (int d = 0; d < dim; ++d)
103 {
104 cell_velocity_fluctuation_squared_sum[d] +=
105 (particle_properties[DEM::PropertiesIndex::v_x + d] -
106 velocity_in_cell_average[d]) *
107 (particle_properties[DEM::PropertiesIndex::v_x + d] -
108 velocity_in_cell_average[d]);
109 }
110
111 particles_cell_number++;
112 }
113 // Calculate average granular temperature in the cell
114 for (int d = 0; d < dim; ++d)
115 {
116 cell_velocity_fluctuation_squared_average[d] =
117 cell_velocity_fluctuation_squared_sum[d] /
118 particles_cell_number;
119 granular_temperature_cell +=
120 (1.0 / dim) * cell_velocity_fluctuation_squared_average[d];
121 }
122 }
123 granular_temperature_average[cell->active_cell_index()] =
124 granular_temperature_cell;
125 }
126 }
127}

129template <int dim>
130Tensor<1, dim>
131calculate_cell_average_particles_velocity(
132 const typename parallel::distributed::Triangulation<dim>::cell_iterator &cell,
133 const Particles::ParticleHandler<dim> &particle_handler)
134{
135 Tensor<1, dim> velocity_cell_sum = Tensor<1, dim>();
136 Tensor<1, dim> velocity_cell_average = Tensor<1, dim>();
137 unsigned int particles_cell_number(0);
138
139 // Looping through all the particles in the cell
140 // Particles in the cell
141 typename Particles::ParticleHandler<dim>::particle_iterator_range
142 particles_in_cell = particle_handler.particles_in_cell(cell);
143
144 const bool particles_exist_in_main_cell = !particles_in_cell.empty();
145
146 // Check to see if the cell has any particles
147 if (particles_exist_in_main_cell)
148 {
149 for (typename Particles::ParticleHandler<dim>::particle_iterator_range::
150 iterator particles_in_cell_iterator = particles_in_cell.begin();
151 particles_in_cell_iterator != particles_in_cell.end();
152 ++particles_in_cell_iterator)
153 {
154 auto particle_properties =
155 particles_in_cell_iterator->get_properties();
156
157 for (int d = 0; d < dim; ++d)
158 {
159 velocity_cell_sum[d] +=
160 particle_properties[DEM::PropertiesIndex::v_x + d];
161 }
162
163 particles_cell_number++;
164 }
165 // Calculate average velocity in the cell
166 for (int d = 0; d < dim; ++d)
167 velocity_cell_average[d] = velocity_cell_sum[d] / particles_cell_number;
168 }
169 return velocity_cell_average;
170}

172template <int dim>
173void
174write_post_processing_results(
175 const parallel::distributed::Triangulation<dim> &triangulation,
176 PVDHandler &grid_pvdhandler,
177 const DoFHandler<dim> &background_dh,
178 const Particles::ParticleHandler<dim> &particle_handler,
179 const DEMSolverParameters<dim> &dem_parameters,
180 const double current_time,
181 const unsigned int step_number,
182 const MPI_Comm &mpi_communicator,
183 AdaptiveSparseContacts<dim> &sparse_contacts_object)
184{
185 const std::string folder = dem_parameters.simulation_control.output_folder;
186 const std::string particles_solution_name =
187 dem_parameters.simulation_control.output_name;
188 const unsigned int group_files =
189 dem_parameters.simulation_control.group_files;
190
191 DataOut<dim> data_out;
192 data_out.attach_dof_handler(background_dh);
193
194 // Write particles' average velocity
195 Vector<double> velocity_average_x(triangulation.n_active_cells());
196 Vector<double> velocity_average_y(triangulation.n_active_cells());
197 Vector<double> velocity_average_z(triangulation.n_active_cells());
198 Vector<double> velocity_average_magnitude(triangulation.n_active_cells());
199 calculate_average_particles_velocity<dim>(triangulation,
200 particle_handler,
201 velocity_average_x,
202 velocity_average_y,
203 velocity_average_z,
204 velocity_average_magnitude);
205
206 data_out.add_data_vector(velocity_average_x,
207 "average_velocity_x",
208 DataOut<dim>::type_cell_data);
209 data_out.add_data_vector(velocity_average_y,
210 "average_velocity_y",
211 DataOut<dim>::type_cell_data);
212
213 if constexpr (dim == 3)
214 data_out.add_data_vector(velocity_average_z,
215 "average_velocity_z",
216 DataOut<dim>::type_cell_data);
217
218 data_out.add_data_vector(velocity_average_magnitude,
219 "average_velocity_magnitude",
220 DataOut<dim>::type_cell_data);
221
222 // Write particles' granular temperature
223 Vector<double> granular_temperature_average(triangulation.n_active_cells());
224 calculate_average_granular_temperature<dim>(triangulation,
225 particle_handler,
226 granular_temperature_average);
227
228 data_out.add_data_vector(granular_temperature_average,
229 "granular_temperature",
230 DataOut<dim>::type_cell_data);
231
232 // Write mobility status of cells
233 Vector<float> mobility_status(triangulation.n_active_cells());
234 sparse_contacts_object.get_mobility_status_vector(mobility_status);
235 data_out.add_data_vector(mobility_status,
236 "mobility_status",
237 DataOut<dim>::type_cell_data);
238
239 // Attach the solution data to data_out object
240 Vector<float> subdomain(triangulation.n_active_cells());
241 for (unsigned int i = 0; i < subdomain.size(); ++i)
242 subdomain(i) = triangulation.locally_owned_subdomain();
243 data_out.add_data_vector(subdomain, "subdomain");
244
245 const std::string postprocess_file_name =
246 dem_parameters.simulation_control.output_name +
247 "_lagrangian_postprocessing";
248
249 data_out.build_patches();
250
251 write_vtu_and_pvd<dim>(grid_pvdhandler,
252 data_out,
253 folder,
254 postprocess_file_name,
255 current_time,
256 step_number,
257 group_files,
258 mpi_communicator);
259}
