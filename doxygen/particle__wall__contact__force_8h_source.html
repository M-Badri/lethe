109 /**
110 * @brief This function is used to find the projection of vector_a on
111 * vector_b
112 * @param vector_a A vector which is going to be projected on vector_b
113 * @param vector_b The projection vector of vector_a
114 * @return The projection of vector_a on vector_b
115 */
116 inline Tensor<1, 3>
117 find_projection(const Tensor<1, 3> &vector_a, const Tensor<1, 3> &vector_b)
118 {
119 Tensor<1, 3> vector_c;
120 vector_c = ((vector_a * vector_b) / (vector_b.norm_square())) * vector_b;
121
122 return vector_c;
123 }

164 /**
165 * @brief Apply the calculated force and torque on the particle in
166 * contact with the given wall, for both non-linear and linear contact force
167 * calculations
168 *
169 * @param forces_and_torques A tuple which contains: 1, normal force, 2,
170 * tangential force, 3, tangential torque and 4, rolling resistance torque of
171 * a contact pair
172 * @param particle_torque Torque acting on particle
173 * @param particle_force Force acting on particle
174 * @param point_on_boundary Contact point on the wall
175 * @param boundary_id ID of the boundary
176 */
177 inline void
178 apply_force_and_torque(
179 const std::tuple<Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>>
180 &forces_and_torques,
181 Tensor<1, 3> &particle_torque,
182 Tensor<1, 3> &particle_force,
183 const Point<3> &point_on_boundary,
184 int boundary_id = 0)
185 {
186 // Getting the values from the forces_and_torques tuple, which are: 1,
187 // normal force, 2, tangential force, 3, tangential torque and 4, rolling
188 // resistance torque
189 Tensor<1, 3> normal_force = std::get<0>(forces_and_torques);
190 Tensor<1, 3> tangential_force = std::get<1>(forces_and_torques);
191 Tensor<1, 3> tangential_torque = std::get<2>(forces_and_torques);
192 Tensor<1, 3> rolling_resistance_torque = std::get<3>(forces_and_torques);
193
194 // Calculation of total force
195 Tensor<1, 3> total_force = normal_force + tangential_force;
196
197 calculate_force_and_torque_on_boundary(boundary_id,
198 total_force,
199 point_on_boundary);
200
201 // Updating the force of particles in the particle handler
202 // Since the forces were calculated on the wall, we use the -= operator
203 particle_force -= total_force;
204
205 // Updating the torque acting on particles
206 // The torque was direcly calculated on the particle, thus we use the +=
207 // operator
208 particle_torque += tangential_torque + rolling_resistance_torque;
209 }
