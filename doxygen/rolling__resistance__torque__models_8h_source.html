34/**
35 * @brief Calculation of the constant rolling resistance torque model using the
36 * information obtained from the fine search and physical properties.
37 *
38 * The model woks as follows:
39 * M_r = - mu_r * R_eff * |F_n| * omega_hat
40 * omega_hat = (omega_i - omega_j) / (|oemaga_i - omega_j|)
41 *
42 * @tparam PropertiesIndex Index of the properties used within the ParticleHandler.
43 *
44 * @param[in] effective_r Effective radius.
45 * @param[in] particle_one_properties Properties of particle one in contact.
46 * @param[in] particle_two_properties Properties of particle two in contact.
47 * @param[in] effective_rolling_friction_coefficient Effective_rolling friction
48 * coefficient
49 * @param[in] normal_force_norm Norm of the nomal force.
50 *
51 */
52template <typename PropertiesIndex>
53inline Tensor<1, 3>
54constant_rolling_resistance_torque(
55 const double effective_r,
56 const ArrayView<const double> &particle_one_properties,
57 const ArrayView<const double> &particle_two_properties,
58 const double effective_rolling_friction_coefficient,
59 const double normal_force_norm,
60 const Tensor<1, 3> & /*normal_contact_vector*/)
61
62{
63 // For calculation of rolling resistance torque, we need to obtain
64 // omega_ij using rotational velocities of particles one and two
65 Tensor<1, 3> particle_one_angular_velocity, particle_two_angular_velocity;
66 for (int d = 0; d < 3; ++d)
67 {
68 particle_one_angular_velocity[d] =
69 particle_one_properties[PropertiesIndex::omega_x + d];
70 particle_two_angular_velocity[d] =
71 particle_two_properties[PropertiesIndex::omega_x + d];
72 }
73
74 Tensor<1, 3> omega_ij =
75 particle_one_angular_velocity - particle_two_angular_velocity;
76 Tensor<1, 3> omega_ij_direction = omega_ij / (omega_ij.norm() + DBL_MIN);
77
78 // Calculation of rolling resistance torque
79 return (-effective_rolling_friction_coefficient * effective_r *
80 normal_force_norm * omega_ij_direction);
81}

83/**
84 * @brief Calculation of the viscous rolling resistance torque model using the
85 * information obtained from the fine search and physical properties.
86 *
87 * The model woks as follows:
88 * M_r = - mu_r * R_eff * |F_n| * |V_omega| * omega_hat
89 * omega_hat = (omega_i - omega_j) / (|oemaga_i - omega_j|)
90 * V_omega = omega_i × (R_i * n_ij) - omega_j × (R_j * n_ji)
91 *
92 * @tparam PropertiesIndex Index of the properties used within the ParticleHandler.
93 *
94 * @param[in] effective_r Effective radius.
95 * @param[in] particle_one_properties Properties of particle one in contact.
96 * @param[in] particle_two_properties Properties of particle two in contact.
97 * @param[in] effective_rolling_friction_coefficient Effective_rolling friction
98 * coefficient
99 * @param[in] normal_force_norm Norm of the nomal force.
100 * @param[in] normal_contact_vector Normal unit vector.
101 */
102
103template <typename PropertiesIndex>
104inline Tensor<1, 3>
105viscous_rolling_resistance_torque(
106 const double effective_r,
107 const ArrayView<const double> &particle_one_properties,
108 const ArrayView<const double> &particle_two_properties,
109 const double effective_rolling_friction_coefficient,
110 const double normal_force_norm,
111 const Tensor<1, 3> &normal_contact_vector)
112
113{
114 // For calculation of rolling resistance torque, we need to obtain
115 // omega_ij using rotational velocities of particles one and two
116 Tensor<1, 3> particle_one_angular_velocity, particle_two_angular_velocity;
117 for (int d = 0; d < 3; ++d)
118 {
119 particle_one_angular_velocity[d] =
120 particle_one_properties[PropertiesIndex::omega_x + d];
121 particle_two_angular_velocity[d] =
122 particle_two_properties[PropertiesIndex::omega_x + d];
123 }
124
125 Tensor<1, 3> omega_ij =
126 particle_one_angular_velocity - particle_two_angular_velocity;
127 Tensor<1, 3> omega_ij_direction = omega_ij / (omega_ij.norm() + DBL_MIN);
128
129 Tensor<1, 3> v_omega =
130 cross_product_3d(particle_one_angular_velocity,
131 particle_one_properties[PropertiesIndex::dp] * 0.5 *
132 normal_contact_vector) -
133 cross_product_3d(particle_two_angular_velocity,
134 particle_two_properties[PropertiesIndex::dp] * 0.5 *
135 -normal_contact_vector);
136
137 // Calculation of rolling resistance torque
138 return (-effective_rolling_friction_coefficient * effective_r *
139 normal_force_norm * v_omega.norm() * omega_ij_direction);
140}
