12template <int dim>
13std::shared_ptr<ParticleParticleContactForceBase<dim>>
14set_particle_particle_contact_force_model(
15 const DEMSolverParameters<dim> &dem_parameters)
16{
17 ParticleParticleContactForceModel particle_particle_contact_force_model =
18 dem_parameters.model_parameters.particle_particle_contact_force_model;
19
20 std::shared_ptr<ParticleParticleContactForceBase<dim>>
21 particle_particle_contact_force_object;
22
23 switch (particle_particle_contact_force_model)
24 {
25 case ParticleParticleContactForceModel::linear:
26 {
27 set_rolling_resistance_model<
28 dim,
29 ParticleParticleContactForceModel::linear>(
30 dem_parameters, particle_particle_contact_force_object);
31 break;
32 }
33 case ParticleParticleContactForceModel::hertz_mindlin_limit_overlap:
34 {
35 set_rolling_resistance_model<
36 dim,
37 ParticleParticleContactForceModel::hertz_mindlin_limit_overlap>(
38 dem_parameters, particle_particle_contact_force_object);
39 break;
40 }
41 case ParticleParticleContactForceModel::hertz_mindlin_limit_force:
42 {
43 set_rolling_resistance_model<
44 dim,
45 ParticleParticleContactForceModel::hertz_mindlin_limit_force>(
46 dem_parameters, particle_particle_contact_force_object);
47 break;
48 }
49 case ParticleParticleContactForceModel::hertz:
50 {
51 set_rolling_resistance_model<
52 dim,
53 ParticleParticleContactForceModel::hertz>(
54 dem_parameters, particle_particle_contact_force_object);
55 break;
56 }
57 case ParticleParticleContactForceModel::hertz_JKR:
58 {
59 set_rolling_resistance_model<
60 dim,
61 ParticleParticleContactForceModel::hertz_JKR>(
62 dem_parameters, particle_particle_contact_force_object);
63 break;
64 }
65 case ParticleParticleContactForceModel::DMT:
66 {
67 set_rolling_resistance_model<dim,
68 ParticleParticleContactForceModel::DMT>(
69 dem_parameters, particle_particle_contact_force_object);
70 break;
71 }
72 default:
73 {
74 throw std::runtime_error(
75 "The chosen particle-particle contact force model is invalid");
76 break;
77 }
78 }
79 return particle_particle_contact_force_object;
80}

83template <int dim,
84 Parameters::Lagrangian::ParticleParticleContactForceModel
85 particle_particle_contact_force_model>
86void
87set_rolling_resistance_model(
88 const DEMSolverParameters<dim> &dem_parameters,
89 std::shared_ptr<ParticleParticleContactForceBase<dim>>
90 &particle_particle_contact_force_object)
91{
92 RollingResistanceMethod rolling_resistance_method =
93 dem_parameters.model_parameters.rolling_resistance_method;
94
95 switch (rolling_resistance_method)
96 {
97 case RollingResistanceMethod::no_resistance:
98 {
99 particle_particle_contact_force_object =
100 std::make_shared<ParticleParticleContactForce<
101 dim,
102 particle_particle_contact_force_model,
103 RollingResistanceMethod::no_resistance>>(dem_parameters);
104 break;
105 }
106 case RollingResistanceMethod::constant_resistance:
107 {
108 particle_particle_contact_force_object =
109 std::make_shared<ParticleParticleContactForce<
110 dim,
111 particle_particle_contact_force_model,
112 RollingResistanceMethod::constant_resistance>>(dem_parameters);
113 break;
114 }
115 case RollingResistanceMethod::viscous_resistance:
116 {
117 particle_particle_contact_force_object =
118 std::make_shared<ParticleParticleContactForce<
119 dim,
120 particle_particle_contact_force_model,
121 RollingResistanceMethod::viscous_resistance>>(dem_parameters);
122 break;
123 }
124 default:
125 throw std::runtime_error("Invalid rolling resistance method");
126 }
127}
