11 template <int dim>
12 void
13 VOFLinearSubequationsSolver<dim>::setup_dofs()
14 {
15  // Get MPI communicator
16  auto mpi_communicator = this->triangulation->get_communicator();
17 
18  // Distribute and renumber DoFs
19  this->dof_handler.distribute_dofs(*this->fe);
20  DoFRenumbering::Cuthill_McKee(this->dof_handler);
21 
22  // Get locally owned and relevant DoFs
23  this->locally_owned_dofs = this->dof_handler.locally_owned_dofs();
24  this->locally_relevant_dofs =
25  DoFTools::extract_locally_relevant_dofs(this->dof_handler);
26 
27  // Constraints
28  this->constraints.clear();
29  this->constraints.reinit(this->locally_relevant_dofs);
30  DoFTools::make_hanging_node_constraints(this->dof_handler, this->constraints);
31  this->constraints.close();
32 
33  // Sparsity pattern
34  DynamicSparsityPattern dsp(this->locally_relevant_dofs);
35  DoFTools::make_sparsity_pattern(this->dof_handler,
36  dsp,
37  this->constraints,
38  false);
39  SparsityTools::distribute_sparsity_pattern(dsp,
40  this->locally_owned_dofs,
41  mpi_communicator,
42  this->locally_relevant_dofs);
43 
44  // Reinitialize system matrix and right-hand side (rhs)
45  this->system_matrix.reinit(this->locally_owned_dofs,
46  this->locally_owned_dofs,
47  dsp,
48  mpi_communicator);
49  this->system_rhs.reinit(this->locally_owned_dofs, mpi_communicator);
50 
51  // Reinitialize solution vectors
52  this->present_solution.reinit(this->locally_owned_dofs,
53  this->locally_relevant_dofs,
54  mpi_communicator);
55  this->evaluation_point = this->present_solution;
56 
57 
58  if (this->subequation_verbosity != Parameters::Verbosity::quiet)
59  {
60  std::string subequation_string =
61  this->subequations_interface->get_subequation_string(
62  this->subequation_id);
63  this->pcout << " Number of " << subequation_string
64  << " degrees of freedom: " << this->dof_handler.n_dofs()
65  << std::endl;
66  }
67 
68  // Provide DoFHandler and solutions to the subequations interface
69  this->subequations_interface->set_dof_handler(this->subequation_id,
70  &this->dof_handler);
71  this->subequations_interface->set_solution(this->subequation_id,
72  &this->present_solution);
73 }

76 template <int dim>
77 void
78 VOFLinearSubequationsSolver<dim>::solve_linear_system_and_update_solution(
79  const bool &is_post_mesh_adaptation)
80 {
81  auto mpi_communicator = this->triangulation->get_communicator();
82 
83  const AffineConstraints<double> &constraints_used = this->constraints;
84 
85  const bool verbose(
86  this->subequation_verbosity != Parameters::Verbosity::quiet &&
87  this->linear_solver_verbosity != Parameters::Verbosity::quiet &&
88  !is_post_mesh_adaptation);
89 
90  if (verbose)
91  {
92  std::string subequation_string =
93  this->subequations_interface->get_subequation_string(
94  this->subequation_id);
95 
96  this->pcout << " -Solving " << subequation_string << ":" << std::endl;
97  }
98 
99  // Set tolerance
100  const double linear_solver_tolerance =
101  this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
102  .minimum_residual;
103 
104  // Solution vector
105  GlobalVectorType completely_distributed_solution(this->locally_owned_dofs,
106  mpi_communicator);
107  completely_distributed_solution = this->present_solution;
108 
109  if (verbose)
110  {
111  this->pcout << " -Tolerance of iterative solver is: "
112  << linear_solver_tolerance << std::endl;
113  }
114 
115  // ILU preconditioner
116  const double ilu_fill =
117  this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
118  .ilu_precond_fill;
119  const double ilu_atol =
120  this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
121  .ilu_precond_atol;
122  const double ilu_rtol =
123  this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
124  .ilu_precond_rtol;
125  TrilinosWrappers::PreconditionILU::AdditionalData preconditionerOptions(
126  ilu_fill, ilu_atol, ilu_rtol, 0);
127 
128  this->ilu_preconditioner =
129  std::make_shared<TrilinosWrappers::PreconditionILU>();
130 
131  this->ilu_preconditioner->initialize(this->system_matrix,
132  preconditionerOptions);
133 
134  // CG solver
135  SolverControl solver_control(
136  this->simulation_parameters.linear_solver.at(PhysicsID::VOF).max_iterations,
137  linear_solver_tolerance,
138  true,
139  true);
140 
141  TrilinosWrappers::SolverCG solver(solver_control);
142 
143  solver.solve(this->system_matrix,
144  completely_distributed_solution,
145  this->system_rhs,
146  *this->ilu_preconditioner);
147 
148  if (verbose)
149  {
150  this->pcout << " -Iterative solver took " << solver_control.last_step()
151  << " steps to reach a residual norm of "
152  << solver_control.last_value() << std::endl;
153  }
154 
155  // Update constraints vector
156  constraints_used.distribute(completely_distributed_solution);
157 
158  // Update solution vectors
159  this->present_solution = completely_distributed_solution;
160  this->evaluation_point = this->present_solution;
161 }
