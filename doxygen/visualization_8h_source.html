21/**
22 * @brief Building patches of particle properties for visualization.
23 * This function is taken from Aspect and dealii and implemented here.
24 */
25template <int dim>
26class Visualization : public dealii::DataOutInterface<0, dim>
27{
28public:
29 Visualization();
30
31 /**
32 * @brief Build the patches of properties of particles for visualization.
33 *
34 * @param particle_handler The particle handler of active particles for
35 * visualization.
36 * @param properties Properties of particles for visualization. This is a
37 * vector of pairs and each pair contains the property name as the first
38 * element and the size of the property as the second element. For vectors
39 * only the size of the first element of the vector is defined equal to the
40 * dimension.
41 */
42 void
43 build_patches(Particles::ParticleHandler<dim> &particle_handler,
44 std::vector<std::pair<std::string, int>> properties);
45
46 /**
47 * @brief Print the data of particles in the xyz format.
48 *
49 * @param particle_handler The particle handler of active particles.
50 * @param pcout Printing in parallel.
51 */
52 void
53 print_xyz(dealii::Particles::ParticleHandler<dim> &particle_handler,
54 const MPI_Comm &mpi_communicator,
55 const ConditionalOStream &pcout);
56
57 /**
58 * @brief Print the data of particles in the deal.II intermediate format.
59 *
60 * @param data_to_print The vector of data to be printed.
61 * @param background_dh The DoFHandler of the background grid.
62 * @param mpi_communicator The MPI communicator.
63 */
64 void
65 print_intermediate_format(const Vector<float> &data_to_print,
66 const DoFHandler<dim> &background_dh,
67 const MPI_Comm &mpi_communicator);
68
69 ~Visualization();
70
71private:
72 /**
73 * @brief Implementation of the corresponding function of the base class.
74 */
75 virtual const std::vector<DataOutBase::Patch<0, dim>> &
76 get_patches() const override;
77
78 /**
79 * @brief Implementation of the corresponding function of the base class.
80 */
81 virtual std::vector<std::string>
82 get_dataset_names() const override;
83
84 virtual std::vector<
85 std::tuple<unsigned int,
86 unsigned int,
87 std::string,
88 DataComponentInterpretation::DataComponentInterpretation>>
89 get_nonscalar_data_ranges() const override;
90
91 /**
92 * @brief Output information that is filled by build_patches() and
93 * written by the write function of the base class.
94 */
95 std::vector<DataOutBase::Patch<0, dim>> patches;
96
97 /**

100 std::vector<std::string> dataset_names;

102 /**

110 vector_datasets;
111
112 /**
113 * @brief Particle properties that are written in output files.
114 */
115 std::vector<std::pair<std::string, int>> properties_to_write;
