15/**
16 * @brief IDs associated to the different subequations solved in Lethe.
17 */
18enum class VOFSubequationsID : unsigned int
19{
20 /// VOF phase fraction gradient L2 projection
21 phase_gradient_projection = 0,

27/**
28 * @brief Interface for secondary equations (subequations) solved within
29 * the VOF auxiliary physics.
30 *
31 * @tparam dim Number of dimensions of the problem.
32 */
33template <int dim>
34class VOFSubequationsInterface
35{
36public:
37 /**
38 * @brief Constructor of an interface for subequations that require the use of
39 * a solver within the span of the VOF auxiliary physics' resolution.
40 *
41 * @param[in] p_simulation_parameters Simulation parameters.
42 *
43 * @param[in] p_pcout Parallel cout used to print the information.
44 *
45 * @param[in] p_triangulation Distributed mesh information.
46 *
47 * @param[in] p_multiphysics_interface Multiphysics interface object pointer
48 * used to get information from physics.
49 */
50 VOFSubequationsInterface(
51 const SimulationParameters<dim> &p_simulation_parameters,
52 const ConditionalOStream &p_pcout,
53 std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
54 &p_triangulation,
55 MultiphysicsInterface<dim> *p_multiphysics_interface);
56
57 /**
58 * @brief Default destructor.
59 */
60 ~VOFSubequationsInterface() = default;
61
62 /**
63 * @brief Setup the DofHandler and the degree of freedom associated with the
64 * subequations.
65 */
66 void
67 setup_dofs()
68 {
69 for (const auto &subequation : this->subequations)
70 {
71 subequation.second->setup_dofs();
72 }
73 };
74
75 /**
76 * @brief Setup the DofHandler and the degree of freedom associated with the
77 * physics.
78 *
79 * @param[in] subequation_id Identifier associated with the subequation wished
80 * to solve.
81 */
82 void
83 setup_specific_subequation_dofs(const VOFSubequationsID &subequation_id)
84 {
85 AssertThrow((std::find(this->active_subequations.begin(),
86 this->active_subequations.end(),
87 subequation_id) != this->active_subequations.end()),
88 ExcInternalError());
89
90 this->subequations.find(subequation_id)->second->setup_dofs();
91 };
92
93 /**
94 * @brief Call solving method of active subequations.
95 *
96 * @param[in] is_post_mesh_adaptation Indicates if the equation is being
97 * solved during post_mesh_adaptation(), for verbosity.
98 */
99 void
100 solve(const bool &is_post_mesh_adaptation = false)
101 {
102 for (const auto &subequation : this->subequations)
103 {
104 subequation.second->solve(is_post_mesh_adaptation);
105 }
106 }
107
108 /**
109 * @brief Call solving method for a specific subequation.
110 *
111 * @param[in] is_post_mesh_adaptation Indicates if the equation is being
112 * solved during post_mesh_adaptation(), for verbosity.
113 *
114 * @param[in] subequation_id Identifier associated with the subequation wished
115 * to solve.
116 */
117 void
118 solve_specific_subequation(const VOFSubequationsID &subequation_id,
119 const bool &is_post_mesh_adaptation = false)
120 {
121 AssertThrow((std::find(this->active_subequations.begin(),
122 this->active_subequations.end(),
123 subequation_id) != this->active_subequations.end()),
124 ExcInternalError());
125
126 this->subequations.find(subequation_id)
127 ->second->solve(is_post_mesh_adaptation);
128 }

130 /**
131 * @brief Get vector of active subequations.

133 * @return Vector of active subequations identifiers.

135 std::vector<VOFSubequationsID>
136 get_active_subequations()

141 /**
142 * @brief Get a pointer to the DoFHandler of a specific subequation.

149 DoFHandler<dim> *

160 /**
161 * @brief Get a pointer to the solution vector of a specific subequation.
162 *
163 * @param[in] subequation_id Identifier associated with a specific
164 * subequation.
165 *
166 * @return Pointer to the solution vector of the specified subequation.
167 */
168 GlobalVectorType *
169 get_solution(const VOFSubequationsID &subequation_id)
170 {
171 AssertThrow((std::find(this->active_subequations.begin(),
172 this->active_subequations.end(),
173 subequation_id) != this->active_subequations.end()),
174 ExcInternalError());
175 return this->subequations_solutions[subequation_id];
176 }
177
178 /**
179 * @brief Get the string associated with a specific subequation.
180 *
181 * @param[in] subequation_id Identifier associated with a specific
182 * subequation.
183 *
184 * @return String associated with the specified subequation.
185 *
186 * @note This is used for printing purposes.
187 */
188 std::string
189 get_subequation_string(const VOFSubequationsID &subequation_id)
190 {
191 AssertThrow((std::find(this->active_subequations.begin(),
192 this->active_subequations.end(),
193 subequation_id) != this->active_subequations.end()),
194 ExcInternalError());
195
196 std::string subequation_string;
197
198 if (subequation_id == VOFSubequationsID::phase_gradient_projection)
199 subequation_string = "VOF phase fraction gradient L2 projection";
200 else if (subequation_id == VOFSubequationsID::curvature_projection)
201 subequation_string = "VOF curvature L2 projection";
202
203 return subequation_string;
204 }
205
206 /**
207 * @brief Set the DoFHandler associated with a specific subequation in the
208 * interface.
209 *
210 * @param[in] subequation_id Identifier associated with a specific
211 * subequation.
212 *
213 * @param[in] dof_handler Pointer to the DoFHandler of a specific subequation.
214 */
215 void
216 set_dof_handler(const VOFSubequationsID &subequation_id,
217 DoFHandler<dim> *dof_handler)
218 {
219 AssertThrow((std::find(this->active_subequations.begin(),
220 this->active_subequations.end(),
221 subequation_id) != this->active_subequations.end()),
222 ExcInternalError());
223 this->subequations_dof_handler[subequation_id] = dof_handler;
224 }
225
226 /**
227 * @brief Set the solution associated with a specific subequation in the
228 * interface.
229 *
230 * @param[in] subequation_id Identifier associated with a specific
231 * subequation.
232 *
233 * @param[in] solution_vector Pointer to the solution vector of a specific
234 * subequation.
235 */
236 void
237 set_solution(const VOFSubequationsID &subequation_id,
238 GlobalVectorType *solution_vector)
239 {
240 AssertThrow((std::find(this->active_subequations.begin(),
241 this->active_subequations.end(),
242 subequation_id) != this->active_subequations.end()),
243 ExcInternalError());
244 this->subequations_solutions[subequation_id] = solution_vector;
245 }

247private:
248 MultiphysicsInterface<dim> *multiphysics_interface;

250 const ConditionalOStream pcout;

264};
